[Unit]
Description=Launches a very basic hotspot for emergency access
After=NetworkManager.service
Requires=NetworkManager.service
ConditionPathExists=/boot/firmware/candle_emergency_hotspot.txt

[Service]
Type=oneshot
ExecStart=/usr/bin/nmcli dev wifi hotspot ifname wlan0 ssid "Candle emergency"
RemainAfterExit=yes

[Install]
WantedBy=multi-user.target
